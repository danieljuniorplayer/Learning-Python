#como iniciar um repositorio
dentro da pasta que vc quer usar, botao direito 
-> guit bash here
->git init #cria o arquivo git na pasta

#como adicionar arquivos ao controle de versao
	git add .
#como criar versoes do codigo com commit
git commit -m "descricao"
#configuracoes iniciais
git config --global user.email "seu email aqui"
git config --global user.name "seu usuario aqui"

#enviar alteracoes para nuvem
git remote add origin  "seu link aqui"
git push --set-upstream origin master #envia o arquivo para branch master
git push #envia o arquivo para nuvem

#Resumindo tudo
git status
git add .
git commit -m "descricao"
git push
git reflog 		      #mostra quais versoes foram modificadas
git reset --hard 3h34h3h	#volta uma versao anterior

# O que são branch?
-> caminhos diferentes
git branch	 //mostras quais branch existem
git branch staging //cria uma nova branch chamada staging
git checkout staging //altera para branch staging

#subir a branch para nuvem
git push --set-upstream origin staging

#Subir as informações de uma branch para outra
git checkout master  //Antes, devemos entrar na branch que vai receber os dados da outra 
git merge staging //vai mergir o que tá em staging para a branch atual (master)

#Descendo as atualizações do Servidor para Maquina Local
git pull
